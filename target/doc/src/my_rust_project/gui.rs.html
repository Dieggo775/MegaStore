22// Fun√ß√£o auxiliar para identificar se o caractere √© um acento/diacr√≠tico
23fn is_mark(c: char) -> bool {
24    matches!(
25        c,
26        '\u{0300}'..='\u{036F}'   // Faixa padr√£o de acentos
27        | '\u{1AB0}'..='\u{1AFF}'
28        | '\u{1DC0}'..='\u{1DFF}'
29        | '\u{20D0}'..='\u{20FF}'
30        | '\u{FE20}'..='\u{FE2F}'
31    )
32}
33
34// Fun√ß√£o que remove acentos de uma string usando normaliza√ß√£o
35fn remover_acentos(s: &str) -> String {
36    s.nfd().filter(|c| !is_mark(*c)).collect()
37}
